    async function scanFile() {
      const fileInput = document.getElementById("fileInput");
      const file = fileInput.files[0];
      const wizard = document.getElementById("wizard");
      const verdict = document.getElementById("verdict");
      const scanStatus = document.getElementById("scanStatus");
      const badge = document.getElementById("badge");
      const loreCapsule = document.getElementById("loreCapsule");
      const loreText = document.getElementById("loreText");

      if (!file) {
        scanStatus.textContent = "‚ùå No file selected.";
        return;
      }

      verdict.textContent = "";
      scanStatus.textContent = "üîç Scanning file...";
      badge.style.display = "none";
      loreCapsule.style.display = "none";
      wizard.classList.add("glow");

      const formData = new FormData();
      formData.append("file", file);

      try {
        const response = await fetch("http://localhost:5000/scan", {
          method: "POST",
          body: formData
        });

        if (!response.ok) {
          throw new Error(`Server responded with ${response.status}`);
        }

        const result = await response.json();
        console.log("Scan result:", result);

        wizard.classList.remove("glow");
        verdict.textContent = `${result.verdict}: ${result.filename}`;
        scanStatus.textContent = `SHA256: ${result.sha256}`;
        badge.style.display = "block";
        loreCapsule.style.display = "block";
        loreText.textContent = result.lore;

      } catch (error) {
        console.error("Scan error:", error);
        wizard.classList.remove("glow");
        scanStatus.textContent = "‚ùå Scan failed. Check console for details.";
      }
    }
  </script>
